# OEE 관리기능 — 플랜 2: 입력 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** OEE 리소스·비가동사유 마스터 관리 화면과, 작업자가 근무조 단위로 가동/비가동 일지를 입력·저장하는 화면·API를 구축한다.

**Architecture:** 플랜 1의 테이블·검증 모듈 위에 쿼리 빌더 → write API → 화면 순으로 쌓는다. 근무조 저장은 "해당 근무조 기존 행 DELETE + 신규 INSERT"를 `executeTransaction`으로 원자화한다. 프론트·API 모두 `validateIntervals`(플랜 1)를 공유한다.

**Tech Stack:** Next.js 16 App Router, React 19, TypeScript, `oracledb`, ag-grid-react(마스터 표), localStorage(단말 프로파일).

**선행 조건:** 플랜 1(`2026-07-06-oee-foundation.md`) 완료 — 테이블 5종, `executeTransaction`, `src/lib/oee/{oee-calc,oee-validation,types}.ts`.

## Global Constraints

- Oracle 접근은 `src/lib/db.ts` 헬퍼만 사용. 근무조 원자적 저장은 `executeTransaction`.
- 검증식은 `src/lib/oee/oee-validation.ts`를 프론트·API가 **공유** — 재작성 금지.
- 시간 표현: 근무조 일지의 `startMin`/`endMin`은 **WORK_DATE 00:00 기준 경과분**(야간조는 1440 초과 허용). 저장 시 `START_TIME = TO_DATE(workDate)+startMin/1440`.
- Oracle 오류(`ORA-*`/`NJS-*`) 원문 보존. 검증 실패는 400 + 오류 배열.
- 인증 없음 — `created_by`는 요청 body의 작업자 사번(필수).
- 소스 UTF-8, Edit/Write 도구로 편집.

/** OEE_RESOURCE 쿼리 빌더 */
export function sqlSelectResources(): string {
  return `SELECT RESOURCE_ID, ORGANIZATION_ID, PROCESS_CODE, RESOURCE_TYPE, REF_CODE,
       RESOURCE_NAME, IDEAL_CT, USE_YN, SORT_ORDER
  FROM OEE_RESOURCE
 WHERE USE_YN = 'Y'
 ORDER BY PROCESS_CODE, SORT_ORDER`;
}
export function sqlInsertResource(): string {
  return `INSERT INTO OEE_RESOURCE
    (ORGANIZATION_ID, PROCESS_CODE, RESOURCE_TYPE, REF_CODE, RESOURCE_NAME, IDEAL_CT, USE_YN, SORT_ORDER)
  VALUES
    (:organizationId, :processCode, :resourceType, :refCode, :resourceName, :idealCt, :useYn, :sortOrder)`;
}
export function sqlUpdateResource(): string {
  return `UPDATE OEE_RESOURCE
     SET PROCESS_CODE = :processCode, RESOURCE_TYPE = :resourceType, REF_CODE = :refCode,
         RESOURCE_NAME = :resourceName, IDEAL_CT = :idealCt, USE_YN = :useYn, SORT_ORDER = :sortOrder
   WHERE RESOURCE_ID = :resourceId`;
}

/** OEE_OPERATION_LOG 근무조 단위 쿼리 빌더.
 *  startMin/endMin은 WORK_DATE 00:00 기준 경과분. */
export function sqlSelectShiftLog(): string {
  return `SELECT LOG_ID, RESOURCE_ID, PROCESS_CODE,
       TO_CHAR(WORK_DATE,'YYYY-MM-DD') AS WORK_DATE, SHIFT,
       ROUND((START_TIME - TRUNC(WORK_DATE)) * 1440) AS START_MIN,
       ROUND((END_TIME   - TRUNC(WORK_DATE)) * 1440) AS END_MIN,
       DURATION_MIN, STATUS, REASON_CODE, RUN_NO, REMARK, CREATED_BY
  FROM OEE_OPERATION_LOG
 WHERE RESOURCE_ID = :resourceId
   AND WORK_DATE = TO_DATE(:workDate,'YYYY-MM-DD')
   AND SHIFT = :shift
 ORDER BY START_TIME`;
}
export function sqlDeleteShiftLog(): string {
  return `DELETE FROM OEE_OPERATION_LOG
 WHERE RESOURCE_ID = :resourceId
   AND WORK_DATE = TO_DATE(:workDate,'YYYY-MM-DD')
   AND SHIFT = :shift`;
}
export function sqlInsertLog(): string {
  return `INSERT INTO OEE_OPERATION_LOG
    (ORGANIZATION_ID, RESOURCE_ID, PROCESS_CODE, WORK_DATE, SHIFT,
     START_TIME, END_TIME, DURATION_MIN, STATUS, REASON_CODE, RUN_NO, REMARK, CREATED_BY)
  VALUES
    (:organizationId, :resourceId, :processCode, TO_DATE(:workDate,'YYYY-MM-DD'), :shift,
     TO_DATE(:workDate,'YYYY-MM-DD') + :startMin/1440,
     TO_DATE(:workDate,'YYYY-MM-DD') + :endMin/1440,
     :durationMin, :status, :reasonCode, :runNo, :remark, :createdBy)`;
}
```

/**
 * @file OEE 리소스 마스터 API
 *   GET: 사용중 리소스 목록 / POST: 신규 / PUT: 수정
 */
import { NextResponse } from 'next/server';
import { executeQuery, executeDml } from '@/lib/db';
import { sqlSelectResources, sqlInsertResource, sqlUpdateResource } from '@/lib/queries/oee/resource';

export const dynamic = 'force-dynamic';

interface ResourceBody {
  resourceId?: number;
  organizationId: number;
  processCode: string;
  resourceType: string;
  refCode: string | null;
  resourceName: string;
  idealCt: number | null;
  useYn?: string;
  sortOrder?: number;
}

export async function GET() {
  try {
    const resources = await executeQuery(sqlSelectResources());
    return NextResponse.json({ resources });
  } catch (error) {
    console.error('[API /oee/resource GET]', error);
    return NextResponse.json({ error: 'Database query failed', resources: [] }, { status: 500 });
  }
}

async function upsert(request: Request, isUpdate: boolean) {
  const b = (await request.json()) as ResourceBody;
  if (!b.organizationId || !b.processCode || !b.resourceType || !b.resourceName) {
    return NextResponse.json({ error: 'organizationId, processCode, resourceType, resourceName 은 필수입니다' }, { status: 400 });
  }
  const binds = {
    organizationId: b.organizationId, processCode: b.processCode, resourceType: b.resourceType,
    refCode: b.refCode ?? null, resourceName: b.resourceName, idealCt: b.idealCt ?? null,
    useYn: b.useYn ?? 'Y', sortOrder: b.sortOrder ?? 0,
    ...(isUpdate ? { resourceId: b.resourceId } : {}),
  };
  await executeDml(isUpdate ? sqlUpdateResource() : sqlInsertResource(), binds);
  return NextResponse.json({ ok: true });
}

export async function POST(request: Request) {
  try { return await upsert(request, false); }
  catch (error) { console.error('[API /oee/resource POST]', error); return NextResponse.json({ error: 'Database query failed' }, { status: 500 }); }
}
export async function PUT(request: Request) {
  try { return await upsert(request, true); }
  catch (error) { console.error('[API /oee/resource PUT]', error); return NextResponse.json({ error: 'Database query failed' }, { status: 500 }); }
}
```

import useSWR from 'swr';

const TYPES = ['MACHINE', 'LINE', 'WORKGROUP'];
const fetcher = (u: string) => fetch(u).then((r) => r.json());

interface Row { RESOURCE_ID: number; PROCESS_CODE: string; RESOURCE_TYPE: string; REF_CODE: string | null; RESOURCE_NAME: string; IDEAL_CT: number | null; SORT_ORDER: number; }

export default function OeeResourceMasterPage() {

  const [form, setForm] = useState({ resourceId: 0, organizationId: 1, processCode: 'SMT', resourceType: 'LINE', refCode: '', resourceName: '', idealCt: '', sortOrder: 0 });
  const editing = form.resourceId > 0;

  async function save() {
    if (!form.resourceName) { alert('리소스명은 필수입니다'); return; }
    const body = { ...form, refCode: form.refCode || null, idealCt: form.idealCt ? Number(form.idealCt) : null };

    setForm({ resourceId: 0, organizationId: 1, processCode: 'SMT', resourceType: 'LINE', refCode: '', resourceName: '', idealCt: '', sortOrder: 0 });

        <input placeholder="이론CT(초)" value={form.idealCt} onChange={(e) => setForm({ ...form, idealCt: e.target.value })} className="border p-2 w-28" />

'', resourceName: r.RESOURCE_NAME, idealCt: r.IDEAL_CT?.toString() ?? '', sortOrder: r.SORT_ORDER })} className="text-blue-600">편집</button></td>

import { NextResponse } from 'next/server';
import { executeQuery, executeTransaction } from '@/lib/db';
import { validateIntervals } from '@/lib/oee/oee-validation';
import type { LogInterval } from '@/lib/oee/types';
import { sqlSelectShiftLog, sqlDeleteShiftLog, sqlInsertLog } from '@/lib/queries/oee/operation-log';

export const dynamic = 'force-dynamic';

interface SaveInterval { startMin: number; endMin: number; status: 'RUN' | 'DOWN'; reasonCode: string | null; runNo?: string | null; remark?: string | null; }
interface SaveBody {
  organizationId: number; resourceId: number; processCode: string;
  workDate: string; shift: string; netLoadMinutes: number; createdBy: string;
  intervals: SaveInterval[];
}

export async function GET(request: Request) {
  try {
    const { searchParams } = new URL(request.url);
    const binds = {
      resourceId: Number(searchParams.get('resourceId')),
      workDate: searchParams.get('workDate') ?? '',
      shift: searchParams.get('shift') ?? '',
    };
    const rows = await executeQuery(sqlSelectShiftLog(), binds);
    return NextResponse.json({ rows });
  } catch (error) {
    console.error('[API /oee/log GET]', error);
    return NextResponse.json({ error: 'Database query failed', rows: [] }, { status: 500 });
  }
}

export async function POST(request: Request) {
  try {
    const b = (await request.json()) as SaveBody;
    if (!b.resourceId || !b.workDate || !b.shift || !b.createdBy) {
      return NextResponse.json({ error: 'resourceId, workDate, shift, createdBy 는 필수입니다' }, { status: 400 });
    }
    const intervals: LogInterval[] = b.intervals.map((iv) => ({ startMin: iv.startMin, endMin: iv.endMin, status: iv.status, reasonCode: iv.reasonCode }));
    const errors = validateIntervals(intervals, b.netLoadMinutes);
    if (errors.length > 0) {
      return NextResponse.json({ error: '검증 실패', errors }, { status: 400 });
    }

    const statements: Array<{ sql: string; binds: Record<string, unknown> }> = [
      { sql: sqlDeleteShiftLog(), binds: { resourceId: b.resourceId, workDate: b.workDate, shift: b.shift } },
      ...b.intervals.map((iv) => ({
        sql: sqlInsertLog(),
        binds: {
          organizationId: b.organizationId, resourceId: b.resourceId, processCode: b.processCode,
          workDate: b.workDate, shift: b.shift, startMin: iv.startMin, endMin: iv.endMin,
          durationMin: iv.endMin - iv.startMin, status: iv.status, reasonCode: iv.reasonCode,
          runNo: iv.runNo ?? null, remark: iv.remark ?? null, createdBy: b.createdBy,
        },
      })),
    ];
    await executeTransaction(statements);
    return NextResponse.json({ ok: true });
  } catch (error) {
    console.error('[API /oee/log POST]', error);
    return NextResponse.json({ error: 'Database query failed' }, { status: 500 });
  }
}
```

'use client';
import { useEffect, useState } from 'react';

export interface OeeProfile { resourceId: number; processCode: string; shift: string; }
const KEY = 'oee-terminal-profile';

export function useOeeProfile() {
  const [profile, setProfileState] = useState<OeeProfile | null>(null);
  useEffect(() => {
    const raw = localStorage.getItem(KEY);
    if (raw) setProfileState(JSON.parse(raw) as OeeProfile);
  }, []);
  function setProfile(p: OeeProfile) {
    localStorage.setItem(KEY, JSON.stringify(p));
    setProfileState(p);
  }
  return { profile, setProfile };
}

'use client';
import type { SaveInterval } from '../page';

interface Reason { REASON_CODE: string; REASON_NAME: string; }
interface Props { intervals: SaveInterval[]; reasons: Reason[]; onChange: (next: SaveInterval[]) => void; }

export default function IntervalEditor({ intervals, reasons, onChange }: Props) {
  function update(i: number, patch: Partial<SaveInterval>) {
    onChange(intervals.map((iv, idx) => (idx === i ? { ...iv, ...patch } : iv)));
  }
  function add() { onChange([...intervals, { startMin: 0, endMin: 0, status: 'RUN', reasonCode: null, runNo: null, remark: null }]); }
  function remove(i: number) { onChange(intervals.filter((_, idx) => idx !== i)); }

  return (
    <div className="space-y-2">
      {intervals.map((iv, i) => (
        <div key={i} className="flex flex-wrap gap-2 items-center border p-2 rounded">
          <input type="number" value={iv.startMin} onChange={(e) => update(i, { startMin: Number(e.target.value) })} className="border p-2 w-24" placeholder="시작(분)" />
          <input type="number" value={iv.endMin} onChange={(e) => update(i, { endMin: Number(e.target.value) })} className="border p-2 w-24" placeholder="종료(분)" />
          <select value={iv.status} onChange={(e) => update(i, { status: e.target.value as 'RUN' | 'DOWN', reasonCode: e.target.value === 'RUN' ? null : iv.reasonCode })} className="border p-2">

        </div>

```tsx
'use client';
import { useEffect, useState } from 'react';
import useSWR from 'swr';
import { validateIntervals } from '@/lib/oee/oee-validation';
import { useOeeProfile } from '@/hooks/useOeeProfile';
import IntervalEditor from './_components/IntervalEditor';

export interface SaveInterval { startMin: number; endMin: number; status: 'RUN' | 'DOWN'; reasonCode: string | null; runNo: string | null; remark: string | null; }
const fetcher = (u: string) => fetch(u).then((r) => r.json());

function todayStr() { return new Date().toISOString().slice(0, 10); }

export default function OeeEntryPage() {
  const { profile, setProfile } = useOeeProfile();
  const { data: resData } = useSWR<{ resources: { RESOURCE_ID: number; PROCESS_CODE: string; RESOURCE_NAME: string }[] }>('/api/oee/resource', fetcher);
  const { data: reasonData } = useSWR<{ reasons: { REASON_CODE: string; REASON_NAME: string }[] }>('/api/oee/reason', fetcher);

  const [workDate, setWorkDate] = useState(todayStr());
  const [worker, setWorker] = useState('');
  const [netLoad, setNetLoad] = useState(480);
  const [intervals, setIntervals] = useState<SaveInterval[]>([]);
  const [errorMsg, setErrorMsg] = useState('');

  const resource = resData?.resources.find((r) => r.RESOURCE_ID === profile?.resourceId);

  useEffect(() => {
    if (!profile) return;
    fetch(`/api/oee/log?resourceId=${profile.resourceId}&workDate=${workDate}&shift=${profile.shift}`)
      .then((r) => r.json())
      .then((d: { rows: { START_MIN: number; END_MIN: number; STATUS: 'RUN' | 'DOWN'; REASON_CODE: string | null; RUN_NO: string | null; REMARK: string | null }[] }) => {
        setIntervals(d.rows.map((row) => ({ startMin: row.START_MIN, endMin: row.END_MIN, status: row.STATUS, reasonCode: row.REASON_CODE, runNo: row.RUN_NO, remark: row.REMARK })));
      });
  }, [profile, workDate]);

  async function save() {
    if (!profile || !resource) { setErrorMsg('단말 프로파일을 먼저 설정하세요'); return; }
    if (!worker) { setErrorMsg('작업자 사번을 입력하세요'); return; }
    const errs = validateIntervals(intervals.map((iv) => ({ startMin: iv.startMin, endMin: iv.endMin, status: iv.status, reasonCode: iv.reasonCode })), netLoad);
    if (errs.length > 0) { setErrorMsg(errs.map((e) => e.message).join(' / ')); return; }
    const res = await fetch('/api/oee/log', {
      method: 'POST', headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ organizationId: 1, resourceId: profile.resourceId, processCode: resource.PROCESS_CODE, workDate, shift: profile.shift, netLoadMinutes: netLoad, createdBy: worker, intervals }),
    });
    if (!res.ok) { const j = await res.json(); setErrorMsg(j.error ?? '저장 실패'); return; }
    setErrorMsg('저장 완료');
  }

  if (!profile) {
    return (
      <div className="p-6 space-y-3">
        <h1 className="text-xl font-bold">단말 프로파일 설정</h1>
        <select id="res" className="border p-2">{resData?.resources.map((r) => <option key={r.RESOURCE_ID} value={r.RESOURCE_ID}>{r.PROCESS_CODE} · {r.RESOURCE_NAME}</option>)}</select>
        <select id="shift" className="border p-2"><option value="DAY">주간</option><option value="NIGHT">야간</option></select>
        <button className="bg-blue-600 text-white px-4 py-2 rounded" onClick={() => {
          const rid = Number((document.getElementById('res') as HTMLSelectElement).value);
          const shift = (document.getElementById('shift') as HTMLSelectElement).value;
          const r = resData?.resources.find((x) => x.RESOURCE_ID === rid);
          if (r) setProfile({ resourceId: rid, processCode: r.PROCESS_CODE, shift });
        }}>설정</button>
      </div>
    );
  }

  return (
    <div className="p-6 space-y-4">
      <div className="flex justify-between items-center">
        <h1 className="text-xl font-bold">{resource?.PROCESS_CODE} · {resource?.RESOURCE_NAME} 가동일지</h1>
        <button className="text-sm text-gray-500" onClick={() => setProfile({ resourceId: 0, processCode: '', shift: '' })}>프로파일 변경</button>
      </div>
      <div className="flex flex-wrap gap-3 items-end">
        <label>일자 <input type="date" value={workDate} onChange={(e) => setWorkDate(e.target.value)} className="border p-2" /></label>
        <label>작업자 사번 <input value={worker} onChange={(e) => setWorker(e.target.value)} className="border p-2" placeholder="스캔/입력" /></label>
        <label>계획가동(분) <input type="number" value={netLoad} onChange={(e) => setNetLoad(Number(e.target.value))} className="border p-2 w-24" /></label>
      </div>
      <IntervalEditor intervals={intervals} reasons={reasonData?.reasons ?? []} onChange={setIntervals} />
      {errorMsg && <p className="text-sm text-red-600">{errorMsg}</p>}
      <button onClick={save} className="bg-green-600 text-white px-6 py-2 rounded">저장</button>
    </div>
  );
}
